    Objetivos:

    Objetivo Geral

    Implementar e avaliar um Variational Autoencoder (VAE) no dataset MNIST, compreendendo sua arquitetura, processo de treinamento e capacidades generativas.

    Objetivos Espec√≠ficos

    3. Arquitetura do VAE

    Um Variational Autoencoder (VAE) √© composto por tr√™s componentes principais:

    1. Encoder (q(z|x)): Mapeia a entrada x para os par√¢metros da distribui√ß√£o latente (m√©dia Œº e log-vari√¢ncia log(œÉ¬≤))

    2. Reparameterization Trick: Permite backpropagation atrav√©s de sampling z = Œº + œÉ * Œµ, onde Œµ ~ N(0, I)

    3. Decoder (p(x|z)): Reconstr√≥i a entrada a partir da representa√ß√£o latente z

    Loss Function

    A loss do VAE combina dois termos:

    • Reconstruction Loss: Mede qu√£o bem o decoder reconstr√≥i a entrada
    • KL Divergence: Regulariza o espa√ßo latente para ser similar a N(0, I)

    L = E[log p(x|z)] - KL[q(z|x) || p(z)]

    class VAE(nn.Module):
    def __init__(self, input_dim=784, hidden_dim=400, latent_dim=20):
        """
        Inicializa o Variational Autoencoder.

        Args:
            input_dim (int): Dimens√£o da entrada (784 para MNIST 28x28)
            hidden_dim (int): Dimens√£o da camada oculta
            latent_dim (int): Dimens√£o do espa√ßo latente
        """
        super(VAE, self).__init__()

        self.input_dim = input_dim
        self.hidden_dim = hidden_dim
        self.latent_dim = latent_dim

        self.fc1 = nn.Linear(input_dim, hidden_dim)

        self.fc_mu = nn.Linear(hidden_dim, latent_dim)
        self.fc_logvar = nn.Linear(hidden_dim, latent_dim)

        self.fc3 = nn.Linear(latent_dim, hidden_dim)
        self.fc4 = nn.Linear(hidden_dim, input_dim)

    def encode(self, x):
        """
        Encoder: mapeia entrada x para par√¢metros da distribui√ß√£o latente.

        Args:
            x (torch.Tensor): Entrada com shape (batch_size, input_dim)

        Returns:
            mu (torch.Tensor): M√©dia da distribui√ß√£o latente
            logvar (torch.Tensor): Log-vari√¢ncia da distribui√ß√£o latente
        """
        h = F.relu(self.fc1(x))
        mu = self.fc_mu(h)
        logvar = self.fc_logvar(h)
        return mu, logvar

    def reparameterize(self, mu, logvar):
        """
        Reparameterization Trick: z = Œº + œÉ * Œµ

        Esta t√©cnica permite backpropagation atrav√©s da opera√ß√£o de sampling.
        Ao inv√©s de samplear diretamente de N(Œº, œÉ¬≤), sampleamos Œµ ~ N(0, 1)
        e calculamos z = Œº + œÉ * Œµ.

        Args:
            mu (torch.Tensor): M√©dia
            logvar (torch.Tensor): Log-vari√¢ncia

        Returns:
            z (torch.Tensor): Amostra do espa√ßo latente
        """
        std = torch.exp(0.5 * logvar)  
        eps = torch.randn_like(std)     
        z = mu + eps * std            
        return z

    def decode(self, z):
        """
        Decoder: reconstr√≥i a entrada a partir da representa√ß√£o latente.

        Args:
            z (torch.Tensor): Representa√ß√£o latente

        Returns:
            reconstruction (torch.Tensor): Imagem reconstru√≠da
        """
        h = F.relu(self.fc3(z))
        reconstruction = torch.sigmoid(self.fc4(h)) 
        return reconstruction

    def forward(self, x):
        """
        Forward pass completo do VAE.

        Args:
            x (torch.Tensor): Entrada

        Returns:
            reconstruction (torch.Tensor): Reconstru√ß√£o
            mu (torch.Tensor): M√©dia da distribui√ß√£o latente
            logvar (torch.Tensor): Log-vari√¢ncia da distribui√ß√£o latente
        """
        mu, logvar = self.encode(x)
        z = self.reparameterize(mu, logvar)
        reconstruction = self.decode(z)
        return reconstruction, mu, logvar

    4. Loss Function

    A fun√ß√£o de loss do VAE √© composta por dois termos:

    Reconstruction Loss (BCE):

    • Mede a qualidade da reconstru√ß√£o
    • Usamos Binary Cross-Entropy pois os pixels est√£o em [0, 1]

    KL Divergence Loss:

    • Regulariza o espa√ßo latente
    • For√ßa a distribui√ß√£o q(z|x) a ser pr√≥xima de N(0, I)
    • KL = -0.5 * Œ£(1 + log(œÉ¬≤) - Œº¬≤ - œÉ¬≤)

    A loss total √©: L = Reconstruction Loss + KL Divergence

    def vae_loss(reconstruction, x, mu, logvar):
    """
    Calcula a loss do VAE.

    Args:
        reconstruction (torch.Tensor): Imagem reconstru√≠da
        x (torch.Tensor): Imagem original
        mu (torch.Tensor): M√©dia da distribui√ß√£o latente
        logvar (torch.Tensor): Log-vari√¢ncia da distribui√ß√£o latente

    Returns:
        total_loss (torch.Tensor): Loss total
        bce_loss (torch.Tensor): Reconstruction loss
        kld_loss (torch.Tensor): KL divergence loss
    """
    BCE = F.binary_cross_entropy(reconstruction, x, reduction='sum')

    KLD = -0.5 * torch.sum(1 + logvar - mu.pow(2) - logvar.exp())

    total_loss = BCE + KLD

    return total_loss, BCE, KLD

    5. Treinamento

    Nesta se√ß√£o, treinamos o VAE monitorando:

    • Loss total
    • Reconstruction loss
    • KL divergence loss

    Salvamos tamb√©m algumas reconstru√ß√µes durante o treinamento para visualizar o progresso.

    LEARNING_RATE = 1e-3
NUM_EPOCHS = 20

optimizer = optim.Adam(model.parameters(), lr=LEARNING_RATE)

train_losses = []
train_bce_losses = []
train_kld_losses = []

test_losses = []
test_bce_losses = []
test_kld_losses = []

def train_epoch(model, train_loader, optimizer, epoch):
    """
    Treina o modelo por uma √©poca.
    """
    model.train()
    train_loss = 0
    train_bce = 0
    train_kld = 0

    for batch_idx, (data, _) in enumerate(train_loader):
        data = data.view(-1, IMAGE_SIZE).to(device)

        optimizer.zero_grad()

        reconstruction, mu, logvar = model(data)

        loss, bce, kld = vae_loss(reconstruction, data, mu, logvar)

        loss.backward()

        optimizer.step()

        train_loss += loss.item()
        train_bce += bce.item()
        train_kld += kld.item()

        if batch_idx % 100 == 0:
            print(f'Epoch [{epoch}/{NUM_EPOCHS}] Batch [{batch_idx}/{len(train_loader)}] '
                  f'Loss: {loss.item()/len(data):.4f} '
                  f'BCE: {bce.item()/len(data):.4f} '
                  f'KLD: {kld.item()/len(data):.4f}')

    avg_loss = train_loss / len(train_loader.dataset)
    avg_bce = train_bce / len(train_loader.dataset)
    avg_kld = train_kld / len(train_loader.dataset)

    return avg_loss, avg_bce, avg_kld

def test_epoch(model, test_loader):
    """
    Avalia o modelo no conjunto de teste.
    """
    model.eval()
    test_loss = 0
    test_bce = 0
    test_kld = 0

    with torch.no_grad():
        for data, _ in test_loader:
            data = data.view(-1, IMAGE_SIZE).to(device)
            reconstruction, mu, logvar = model(data)
            loss, bce, kld = vae_loss(reconstruction, data, mu, logvar)

            test_loss += loss.item()
            test_bce += bce.item()
            test_kld += kld.item()

    avg_loss = test_loss / len(test_loader.dataset)
    avg_bce = test_bce / len(test_loader.dataset)
    avg_kld = test_kld / len(test_loader.dataset)

    return avg_loss, avg_bce, avg_kld

print(f"\n{'='*60}")
print(f"Iniciando Treinamento")
print(f"{'='*60}\n")

for epoch in range(1, NUM_EPOCHS + 1):
    train_loss, train_bce, train_kld = train_epoch(model, train_loader, optimizer, epoch)
    train_losses.append(train_loss)
    train_bce_losses.append(train_bce)
    train_kld_losses.append(train_kld)

    test_loss, test_bce, test_kld = test_epoch(model, test_loader)
    test_losses.append(test_loss)
    test_bce_losses.append(test_bce)
    test_kld_losses.append(test_kld)

    print(f'\n==> Epoch {epoch}/{NUM_EPOCHS}:')
    print(f'Train Loss: {train_loss:.4f} (BCE: {train_bce:.4f}, KLD: {train_kld:.4f})')
    print(f'Test Loss: {test_loss:.4f} (BCE: {test_bce:.4f}, KLD: {test_kld:.4f})\n')

print(f"{'='*60}")
print(f"Treinamento Conclu√≠do!")
print(f"{'='*60}\n")
